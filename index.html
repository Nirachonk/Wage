📖 คู่มือตั้งค่า Google Apps Script

ทำครั้งเดียว ข้อมูลจะเก็บใน Google Sheets ของคุณตลอดไป

  1. เปิด sheets.new สร้าง Google Sheets ใหม่ ตั้งชื่อว่า "TimeTrack DB"
  2. คลิกเมนู Extensions → Apps Script
  3. ลบโค้ดเดิมออก แล้ววาง โค้ด Apps Script ด้านล่างนี้ทั้งหมด
  4. กด 💾 Save แล้วกด ▶ Deploy → New deployment
  5. เลือก Type: Web App, Execute as: Me, Who has access: Anyone
  6. กด Deploy → Authorize access → คัดลอก Web App URL
  7. นำ URL มาวางในช่องด้านบนของเว็บนี้ แล้วกด "บันทึก URL"
  8. ⚠️ หากเคย Deploy ไปแล้วและพบปัญหาวันที่/เวลาผิด: ให้เปิด Apps Script → ลบโค้ดเก่า → วางโค้ดใหม่ด้านล่าง → กด Save → Deploy → New deployment (ไม่ใช่ Manage deployments) → คัดลอก URL ใหม่มาใส่
📋 โค้ด Apps Script (คัดลอกทั้งหมด)
const SHEET_NAME = 'entries';

function doGet(e) {
  return handleRequest(e);
}

function doPost(e) {
  return handleRequest(e);
}

// แปลงค่าใน cell ให้เป็น string ที่ถูกต้อง
// Google Sheets จะแปลง date/time เป็น Date object อัตโนมัติ
function cellToString(val, colName) {
  if (val === null || val === undefined || val === '') return '';

  // ถ้าเป็น Date object
  if (val instanceof Date) {
    if (colName === 'date') {
      // ส่งกลับเป็น YYYY-MM-DD
      const y = val.getFullYear();
      const m = String(val.getMonth() + 1).padStart(2, '0');
      const d = String(val.getDate()).padStart(2, '0');
      return y + '-' + m + '-' + d;
    }
    if (colName === 'start' || colName === 'end') {
      // ส่งกลับเป็น HH:MM
      // Google Sheets เก็บเวลาเป็น fraction ของวัน (0-1)
      // แต่ถ้าเป็น Date object ให้ดึง hours/minutes ออกมา
      const h = String(val.getHours()).padStart(2, '0');
      const min = String(val.getMinutes()).padStart(2, '0');
      return h + ':' + min;
    }
    return val.toISOString();
  }

  // ถ้าเป็น number และ column เป็น start/end (Google Sheets เก็บเวลาเป็น fraction 0-1)
  if (typeof val === 'number' && (colName === 'start' || colName === 'end')) {
    const totalMins = Math.round(val * 24 * 60);
    const h = String(Math.floor(totalMins / 60)).padStart(2, '0');
    const m = String(totalMins % 60).padStart(2, '0');
    return h + ':' + m;
  }

  return String(val);
}

function handleRequest(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(['id','name','date','start','end','wage','createdAt']);
    // กำหนดให้ column date, start, end เป็น Plain text เพื่อป้องกัน auto-convert
    sheet.getRange('C:C').setNumberFormat('@STRING@');
    sheet.getRange('D:D').setNumberFormat('@STRING@');
    sheet.getRange('E:E').setNumberFormat('@STRING@');
  }

  const p = e.parameter || {};
  const action = p.action || (e.postData ? JSON.parse(e.postData.contents).action : 'get');
  let body;

  try {
    if (action === 'get') {
      const rows = sheet.getDataRange().getValues();
      const headers = rows[0];
      const data = rows.slice(1).map(r => {
        const obj = {};
        headers.forEach((h, i) => {
          obj[h] = cellToString(r[i], h);
        });
        return obj;
      }).filter(r => r.id && r.id !== '');
      body = { ok: true, data };

    } else if (action === 'add') {
      const d = JSON.parse(e.postData.contents);
      const id = Utilities.getUuid();
      const rowData = [id, d.name, d.date, d.start, d.end, Number(d.wage), new Date().toISOString()];
      sheet.appendRow(rowData);
      // Force format ให้ cell date/start/end เป็น plain text
      const lastRow = sheet.getLastRow();
      sheet.getRange(lastRow, 3).setNumberFormat('@STRING@').setValue(d.date);
      sheet.getRange(lastRow, 4).setNumberFormat('@STRING@').setValue(d.start);
      sheet.getRange(lastRow, 5).setNumberFormat('@STRING@').setValue(d.end);
      body = { ok: true, id };

    } else if (action === 'update') {
      const d = JSON.parse(e.postData.contents);
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(d.id)) {
          sheet.getRange(i+1, 2).setValue(d.name);
          sheet.getRange(i+1, 3).setNumberFormat('@STRING@').setValue(d.date);
          sheet.getRange(i+1, 4).setNumberFormat('@STRING@').setValue(d.start);
          sheet.getRange(i+1, 5).setNumberFormat('@STRING@').setValue(d.end);
          sheet.getRange(i+1, 6).setValue(Number(d.wage));
          break;
        }
      }
      body = { ok: true };

    } else if (action === 'delete') {
      const d = JSON.parse(e.postData.contents);
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(d.id)) {
          sheet.deleteRow(i + 1);
          break;
        }
      }
      body = { ok: true };

    } else {
      body = { ok: false, error: 'Unknown action: ' + action };
    }
  } catch(err) {
    body = { ok: false, error: err.message };
  }

  return ContentService
    .createTextOutput(JSON.stringify(body))
    .setMimeType(ContentService.MimeType.JSON);
}
